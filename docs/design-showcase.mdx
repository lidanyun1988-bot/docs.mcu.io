## 代码块示例

```typescript
// 初始化 MCU 时钟系统
function initClockSystem() {
  // 配置 PLL 参数
  const pllConfig = {
    source: 'HSE',
    multiplier: 168,
    divider: 8
  };
  
  // 启用 Flash 预取
  FLASH.ACR |= FLASH_ACR_PRFTEN;
  
  // 设置等待周期
  FLASH.ACR |= FLASH_ACR_LATENCY_5WS;
  
  console.log('Clock system initialized');
}
```
